        // --- Save/Load Functionality ---
        function saveProject(projectName) {
            const projectData = {};
            for (const lang in editors) {
                projectData[lang] = editors[lang].value;
            }
            localStorage.setItem(`codelab_project_${projectName}`, JSON.stringify(projectData));
            logToConsole(`Project "${projectName}" saved!`, 'success');
            updateProjectList();
        }

        function loadProject(projectName) {
            const projectData = JSON.parse(localStorage.getItem(`codelab_project_${projectName}`));
            if (projectData) {
                for (const lang in editors) {
                    editors[lang].value = projectData[lang] || '';
                }
                // Automatically switch to HTML tab after loading if it exists
                if (projectData.html) {
                    tabButtons.forEach(btn => {
                        if (btn.dataset.lang === 'html') btn.click();
                    });
                } else {
                     // If no HTML, just re-activate the current tab to refresh content
                     tabButtons.forEach(btn => {
                        if (btn.dataset.lang === activeLanguage) btn.click();
                    });
                }
                logToConsole(`Project "${projectName}" loaded!`, 'success');
            } else {
                logToConsole(`Project "${projectName}" not found.`, 'error');
            }
        }

        function deleteProject(projectName) {
            if (confirm(`Are you sure you want to delete "${projectName}"?`)) {
                localStorage.removeItem(`codelab_project_${projectName}`);
                logToConsole(`Project "${projectName}" deleted.`, 'info');
                updateProjectList();
            }
        }

        function updateProjectList() {
            projectList.innerHTML = '';
            let projectsFound = false;
            for (let i = 0; i < localStorage.length; i++) {
                const key = localStorage.key(i);
                if (key.startsWith('codelab_project_')) {
                    projectsFound = true;
                    const projectName = key.replace('codelab_project_', '');
                    const listItem = document.createElement('li');
                    listItem.className = 'flex justify-between items-center bg-gray-700 p-2 rounded-md mb-2 last:mb-0';
                    listItem.innerHTML = `
                        ${projectName}
                        

                            
                            
                        

                    `;
                    projectList.appendChild(listItem);
                }
            }
            if (!projectsFound) {
                projectList.innerHTML = '
  • No saved projects yet.
    • ';
            }

            // Add event listeners to newly created buttons
            projectList.querySelectorAll('.load-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    loadProject(e.target.dataset.project);
                    saveLoadModal.classList.add('hidden'); // Close modal after loading
                });
            });
            projectList.querySelectorAll('.delete-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    deleteProject(e.target.dataset.project);
                });
            });
        }
